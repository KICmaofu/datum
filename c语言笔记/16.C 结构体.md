在C语言中，**结构体（Struct）** 是一种用户自定义的数据类型，用于将**不同类型的数据（如int、float、char等）组合成一个整体**，以表示一个“复合对象”。例如，用结构体表示“学生”（包含学号、姓名、成绩等）、“点”（包含x坐标、y坐标）等，使数据组织更符合现实逻辑。


### 一、结构体的定义
结构体通过 `struct` 关键字定义，基本格式如下：  
```c
struct 结构体名 {
    数据类型 成员1;  // 结构体中的成员（可以是任意数据类型，包括其他结构体）
    数据类型 成员2;
    // ... 更多成员
};
```

- **结构体名**：遵循标识符命名规则（如`Student`、`Point`），用于标识该结构体类型。  
- **成员**：结构体包含的具体数据项，每个成员有自己的类型和名称。  


#### 示例：定义一个“学生”结构体
```c
// 定义结构体Student，包含3个成员
struct Student {
    int id;         // 学号（整数）
    char name[20];  // 姓名（字符数组，即字符串）
    float score;    // 成绩（浮点数）
};
```


### 二、结构体变量的声明与初始化
定义结构体类型后，需声明结构体变量才能使用。声明和初始化有多种方式：


#### 1. 先定义结构体类型，再声明变量
```c
// 定义结构体类型
struct Point {
    int x;  // x坐标
    int y;  // y坐标
};

// 声明结构体变量（p1、p2是struct Point类型的变量）
struct Point p1;
struct Point p2;
```


#### 2. 定义结构体类型的同时声明变量
```c
// 定义结构体Circle，同时声明变量c1、c2
struct Circle {
    int x;      // 圆心x坐标
    int y;      // 圆心y坐标
    float r;    // 半径
} c1, c2;
```


#### 3. 初始化结构体变量
结构体变量可以在声明时初始化（按成员顺序赋值），或声明后逐个给成员赋值。  


##### （1）声明时初始化（按成员顺序）
```c
#include <stdio.h>

struct Student {
    int id;
    char name[20];
    float score;
};

int main() {
    // 初始化：按成员顺序赋值（id=1001, name="Alice", score=95.5）
    struct Student s1 = {1001, "Alice", 95.5};
    return 0;
}
```


##### （2）声明后逐个成员赋值（通过`.`运算符访问成员）
```c
int main() {
    struct Student s2;
    s2.id = 1002;                  // 给id成员赋值
    strcpy(s2.name, "Bob");        // 给字符数组成员赋值（不能直接用=，需用strcpy）
    s2.score = 88.0;               // 给score成员赋值
    return 0;
}
```

**注意**：结构体中的字符数组成员（如`name`）不能直接用`=`赋值（如`s2.name = "Bob"`是错误的），需用字符串复制函数`strcpy`（需包含`#include <string.h>`）。


### 三、结构体成员的访问
访问结构体变量的成员需使用**成员访问运算符**：  
- 对于普通结构体变量：`变量名.成员名`（点运算符`.`）。  
- 对于结构体指针变量：`指针名->成员名`（箭头运算符`->`），等价于`(*指针名).成员名`。  


#### 示例：访问结构体成员
```c
#include <stdio.h>
#include <string.h>

struct Student {
    int id;
    char name[20];
    float score;
};

int main() {
    // 普通变量：用.访问
    struct Student s = {1001, "Alice", 95.5};
    printf("学号：%d\n", s.id);       // 输出：1001
    printf("姓名：%s\n", s.name);     // 输出：Alice
    printf("成绩：%.1f\n", s.score);  // 输出：95.5

    // 结构体指针：用->访问
    struct Student *p = &s;  // p是指向s的指针
    printf("指针访问学号：%d\n", p->id);      // 等价于(*p).id → 1001
    printf("指针访问成绩：%.1f\n", (*p).score);  // 等价于p->score → 95.5

    return 0;
}
```


### 四、结构体数组
结构体数组是**元素为结构体类型的数组**，用于存储多个同类型的复合对象（如多个学生、多个点）。


#### 定义与访问
```c
#include <stdio.h>

struct Student {
    int id;
    char name[20];
    float score;
};

int main() {
    // 定义结构体数组（3个学生）并初始化
    struct Student class[3] = {
        {1001, "Alice", 95.5},
        {1002, "Bob", 88.0},
        {1003, "Charlie", 92.3}
    };

    // 访问数组元素的成员（数组下标 + .运算符）
    for (int i = 0; i < 3; i++) {
        printf("第%d个学生：学号=%d，姓名=%s，成绩=%.1f\n",
               i+1, class[i].id, class[i].name, class[i].score);
    }
    return 0;
}
```

输出：  
```
第1个学生：学号=1001，姓名=Alice，成绩=95.5
第2个学生：学号=1002，姓名=Bob，成绩=88.0
第3个学生：学号=1003，姓名=Charlie，成绩=92.3
```


### 五、结构体作为函数参数
结构体可以作为函数参数传递，有两种方式：**传值调用**和**传址调用**。  


#### 1. 传值调用（复制整个结构体）
函数接收结构体的副本，修改形参不会影响实参（缺点：结构体过大时，复制开销高，效率低）。  

```c
#include <stdio.h>

struct Point {
    int x;
    int y;
};

// 传值调用：接收结构体副本
void printPoint(struct Point p) {
    printf("点坐标：(%d, %d)\n", p.x, p.y);
}

int main() {
    struct Point p = {3, 4};
    printPoint(p);  // 传递p的副本 → 输出：(3, 4)
    return 0;
}
```


#### 2. 传址调用（传递结构体指针）
函数接收结构体的地址（指针），通过指针访问和修改原结构体（优点：无需复制，效率高，可修改实参）。  

```c
#include <stdio.h>

struct Point {
    int x;
    int y;
};

// 传址调用：接收结构体指针
void movePoint(struct Point *p, int dx, int dy) {
    p->x += dx;  // 修改原结构体的x
    p->y += dy;  // 修改原结构体的y
}

int main() {
    struct Point p = {3, 4};
    movePoint(&p, 2, 5);  // 传递p的地址
    printf("移动后坐标：(%d, %d)\n", p.x, p.y);  // 输出：(5, 9)（原结构体被修改）
    return 0;
}
```

**推荐**：优先使用传址调用（指针），尤其是结构体较大时，能显著提高效率。


### 六、typedef 简化结构体使用
每次声明结构体变量时都写`struct 结构体名`（如`struct Student s`）比较繁琐。通过`typedef`为结构体类型定义别名，可以简化声明。  


#### 示例：用typedef定义结构体别名
```c
#include <stdio.h>

// 用typedef定义别名：将struct Student 命名为 Student
typedef struct Student {
    int id;
    char name[20];
    float score;
} Student;  // Student是别名

int main() {
    // 直接用别名声明变量（无需写struct）
    Student s = {1001, "Alice", 95.5};  // 等价于 struct Student s;
    printf("姓名：%s\n", s.name);  // 输出：Alice
    return 0;
}
```

甚至可以省略结构体名，直接定义别名：  
```c
typedef struct {  // 匿名结构体
    int x;
    int y;
} Point;  // 别名为Point

Point p = {3, 4};  // 直接使用别名
```


### 七、结构体的嵌套
结构体的成员可以是另一个结构体（即“结构体嵌套”），用于表示更复杂的复合对象。  


#### 示例：嵌套结构体表示“学生”包含“生日”
```c
#include <stdio.h>

// 定义日期结构体（年、月、日）
typedef struct {
    int year;
    int month;
    int day;
} Date;

// 学生结构体（包含日期结构体作为成员）
typedef struct {
    int id;
    char name[20];
    Date birthday;  // 嵌套结构体：生日
} Student;

int main() {
    // 初始化嵌套结构体
    Student s = {
        1001,
        "Alice",
        {2000, 5, 18}  // 初始化birthday成员（按Date的成员顺序）
    };

    // 访问嵌套结构体的成员（多层.运算符）
    printf("%s的生日：%d年%d月%d日\n",
           s.name,
           s.birthday.year,
           s.birthday.month,
           s.birthday.day);  // 输出：Alice的生日：2000年5月18日
    return 0;
}
```


### 八、注意事项
1. **结构体不能包含自身作为成员**（会导致无限递归，编译错误），但可以包含**自身的指针**（用于实现链表、树等数据结构）。  
   ```c
   // 错误：结构体不能包含自身
   struct Node {
       int data;
       struct Node next;  // 编译错误
   };

   // 正确：可以包含自身指针
   struct Node {
       int data;
       struct Node *next;  // 指针，指向另一个Node（链表节点）
   };
   ```

2. **内存对齐**：结构体的总大小通常不等于各成员大小之和，编译器会自动对齐成员（按最大成员的大小或特定规则），以提高内存访问效率。例如：  
   ```c
   struct A {
       char c;  // 1字节
       int i;   // 4字节
   };
   // sizeof(struct A) 通常为8（而非5）：char后补3字节对齐int
   ```

3. **结构体赋值**：普通结构体变量可以直接赋值（整体复制），但不能直接比较（需手动比较每个成员）。  
   ```c
   Student s1 = {1001, "Alice", 95.5};
   Student s2 = s1;  // 合法：整体复制s1的成员到s2
   // if (s1 == s2) { ... }  // 错误：结构体不能直接用==比较
   ```


### 总结
结构体是C语言中组织复合数据的核心工具，通过将不同类型的数据组合，使代码更贴合现实逻辑。核心要点：  
- 用`struct`定义结构体类型，包含多个成员。  
- 通过`.`（普通变量）或`->`（指针）访问成员。  
- 结构体数组用于存储多个复合对象，传址调用（指针）是函数参数的高效方式。  
- `typedef`可简化结构体变量的声明，嵌套结构体用于表示复杂对象。

   ```c
   struct Node {
       int data;
       struct Node *next;  // 指针，指向另一个Node（链表节点）
   };
   ```

2. **内存对齐**：结构体的总大小通常不等于各成员大小之和，编译器会自动对齐成员（按最大成员的大小或特定规则），以提高内存访问效率。例如：  
   ```c
   struct A {
       char c;  // 1字节
       int i;   // 4字节
   };
   // sizeof(struct A) 通常为8（而非5）：char后补3字节对齐int
   ```

3. **结构体赋值**：普通结构体变量可以直接赋值（整体复制），但不能直接比较（需手动比较每个成员）。  
   ```c
   Student s1 = {1001, "Alice", 95.5};
   Student s2 = s1;  // 合法：整体复制s1的成员到s2
   // if (s1 == s2) { ... }  // 错误：结构体不能直接用==比较
   ```


### 总结
结构体是C语言中组织复合数据的核心工具，通过将不同类型的数据组合，使代码更贴合现实逻辑。核心要点：  
- 用`struct`定义结构体类型，包含多个成员。  
- 通过`.`（普通变量）或`->`（指针）访问成员。  
- 结构体数组用于存储多个复合对象，传址调用（指针）是函数参数的高效方式。  
- `typedef`可简化结构体变量的声明，嵌套结构体用于表示复杂对象。  

结构体是实现链表、树等复杂数据结构的基础，也是模块化编程的重要手段。
