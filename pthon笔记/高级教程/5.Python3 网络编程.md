## 一、网络编程基础概念
Python3 提供了强大的网络编程支持，无需复杂配置即可实现客户端与服务器的通信、HTTP 请求、数据传输等功能。网络编程的核心是 **Socket（套接字）**，它是进程间通信的一种方式，允许不同设备、不同进程通过网络交换数据。

### 核心概念
| 术语         | 说明                                  |
|--------------|---------------------------------------|
| Socket       | 网络通信的端点，包含 IP 地址和端口号  |
| IP 地址      | 设备在网络中的唯一标识（如 `127.0.0.1` 本地回环地址） |
| 端口号       | 区分设备上不同服务的编号（1-65535，0-1023 为系统端口） |
| TCP          | 面向连接、可靠的字节流协议（如 HTTP、文件传输） |
| UDP          | 无连接、不可靠的数据包协议（如视频通话、广播） |
| 客户端/服务器 | 网络通信的两种角色（客户端请求服务，服务器提供服务） |

### Python 网络编程优势
- 内置 `socket` 模块，无需第三方依赖
- 语法简洁，API 直观，学习成本低
- 支持 TCP/UDP、HTTP、FTP 等多种协议
- 可快速实现客户端和服务器端代码
- 适合开发中小型网络应用、接口测试、数据爬取等场景

## 二、Socket 基础：TCP 编程
TCP 是 **面向连接** 的协议，通信前需建立三次握手，通信后需四次挥手断开连接，保证数据可靠传输。适合对数据完整性要求高的场景（如文件传输、登录验证）。

### 1. TCP 服务器（单词查询服务）
以「英语学习平台」的单词查询服务为例，实现简单的 TCP 服务器：
```python
import socket
import threading

# 模拟单词数据库（从之前的 vocabulary 表提取）
word_database = {
    "perseverance": "坚持不懈，毅力",
    "consistent": "一致的，持续的",
    "proficiency": "熟练，精通",
    "fluency": "流利度",
    "diligent": "勤奋的，勤勉的"
}

def handle_client(client_socket, client_addr):
    """处理单个客户端连接（线程函数）"""
    print(f"新连接：{client_addr}")
    try:
        # 发送欢迎消息
        welcome_msg = "欢迎使用英语单词查询服务！输入单词获取释义（输入 'quit' 退出）\n"
        client_socket.send(welcome_msg.encode("utf-8"))
        
        while True:
            # 接收客户端数据（最大 1024 字节）
            data = client_socket.recv(1024).decode("utf-8").strip()
            if not data or data.lower() == "quit":
                print(f"客户端 {client_addr} 断开连接")
                break
            
            # 查询单词并返回结果
            definition = word_database.get(data, f"未找到单词 '{data}' 的释义")
            response = f"单词：{data}\n释义：{definition}\n"
            client_socket.send(response.encode("utf-8"))
    
    except Exception as e:
        print(f"客户端 {client_addr} 通信异常：{e}")
    finally:
        client_socket.close()  # 关闭客户端连接

def start_tcp_server():
    """启动 TCP 服务器"""
    # 创建 TCP Socket（AF_INET：IPv4，SOCK_STREAM：TCP）
    server_socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    
    # 允许端口复用（避免重启服务器时出现 "端口已占用" 错误）
    server_socket.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
    
    # 绑定 IP 和端口（'' 表示监听所有网卡，端口 8888）
    server_socket.bind(('', 8888))
    
    # 开始监听（backlog=5：最大等待连接数）
    server_socket.listen(5)
    print("TCP 单词查询服务器已启动，监听端口 8888...")
    
    try:
        while True:
            # 等待客户端连接（阻塞式）
            client_socket, client_addr = server_socket.accept()
            
            # 开启新线程处理客户端（支持多客户端同时连接）
            client_thread = threading.Thread(target=handle_client, args=(client_socket, client_addr))
            client_thread.daemon = True  # 主线程退出时，子线程自动退出
            client_thread.start()
    
    except KeyboardInterrupt:
        print("\n服务器正在关闭...")
    finally:
        server_socket.close()

if __name__ == "__main__":
    start_tcp_server()
```

### 2. TCP 客户端（单词查询客户端）
实现客户端连接服务器，查询英语单词：
```python
import socket

def tcp_client():
    """TCP 客户端：查询英语单词"""
    # 创建 TCP Socket
    client_socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    
    try:
        # 连接服务器（IP 为服务器地址，端口 8888）
        client_socket.connect(('127.0.0.1', 8888))
        
        # 接收服务器欢迎消息
        welcome_msg = client_socket.recv(1024).decode("utf-8")
        print(welcome_msg)
        
        while True:
            # 输入要查询的单词
            word = input("请输入要查询的单词：").strip()
            if not word:
                print("请输入有效单词！")
                continue
            
            # 发送单词到服务器
            client_socket.send(word.encode("utf-8"))
            
            if word.lower() == "quit":
                print("退出客户端")
                break
            
            # 接收服务器响应
            response = client_socket.recv(1024).decode("utf-8")
            print("="*30)
            print(response)
    
    except ConnectionRefusedError:
        print("连接失败！请确保服务器已启动")
    except Exception as e:
        print(f"客户端异常：{e}")
    finally:
        client_socket.close()

if __name__ == "__main__":
    tcp_client()
```

### 3. 运行说明
1. 先启动服务器代码（`start_tcp_server()`）
2. 再启动一个或多个客户端代码（`tcp_client()`）
3. 客户端输入单词（如 `perseverance`）即可获取释义，输入 `quit` 退出

## 三、Socket 基础：UDP 编程
UDP 是 **无连接** 的协议，无需建立连接直接发送数据包，速度快但不保证数据可靠传输（可能丢失、乱序）。适合对实时性要求高的场景（如语音通话、游戏数据传输）。

### 1. UDP 服务器（英语单词拼写检查）
```python
import socket

# 正确的单词列表
correct_words = {"apple", "banana", "orange", "perseverance", "consistent", "fluency"}

def start_udp_server():
    """UDP 服务器：单词拼写检查"""
    # 创建 UDP Socket（SOCK_DGRAM：UDP）
    server_socket = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
    
    # 绑定 IP 和端口
    server_socket.bind(('', 9999))
    print("UDP 单词拼写检查服务器已启动，监听端口 9999...")
    
    try:
        while True:
            # 接收客户端数据（data：数据，addr：客户端地址）
            data, client_addr = server_socket.recvfrom(1024)
            word = data.decode("utf-8").strip()
            print(f"收到 {client_addr} 的请求：{word}")
            
            # 拼写检查
            if word.lower() == "quit":
                response = "已退出服务"
            elif word in correct_words:
                response = f"✅ 拼写正确：{word}"
            else:
                response = f"❌ 拼写错误：{word}（请检查拼写）"
            
            # 发送响应给客户端
            server_socket.sendto(response.encode("utf-8"), client_addr)
    
    except KeyboardInterrupt:
        print("\n服务器正在关闭...")
    finally:
        server_socket.close()

if __name__ == "__main__":
    start_udp_server()
```

### 2. UDP 客户端（单词拼写检查）
```python
import socket

def udp_client():
    """UDP 客户端：单词拼写检查"""
    # 创建 UDP Socket
    client_socket = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
    
    server_addr = ('127.0.0.1', 9999)
    print("欢迎使用单词拼写检查服务！输入单词检查拼写（输入 'quit' 退出）")
    
    try:
        while True:
            word = input("请输入单词：").strip()
            if not word:
                print("请输入有效单词！")

            client_socket.sendto(word.encode("utf-8"), server_addr)
            
            if word.lower() == "quit":
                print("退出客户端")
                break
            
            # 接收服务器响应
            response, _ = client_socket.recvfrom(1024)
            print("检查结果：", response.decode("utf-8"))
            print("-"*30)
    
    except Exception as e:
        print(f"客户端异常：{e}")
    finally:
        client_socket.close()

if __name__ == "__main__":
    udp_client()
```

### TCP vs UDP 核心区别
| 特性         | TCP                  | UDP                  |
|--------------|----------------------|----------------------|
| 连接方式     | 面向连接（三次握手） | 无连接               |
| 可靠性       | 可靠（数据不丢失、有序） | 不可靠（可能丢失、乱序） |
| 速度         | 较慢（需确认机制）   | 较快（无额外开销）   |
| 数据大小     | 无限制               | 单次传输有限制（通常 <64KB） |
| 适用场景     | 文件传输、登录、HTTP | 视频通话、游戏、广播 |

```python
import requests

def get_english_news():
    """GET 请求：获取简单英语新闻（示例接口）"""
    url = "https://newsapi.org/v2/top-headlines"
    # 请求参数（英语新闻、简单语言）
    params = {
        "language": "en",
        "category": "general",
        "apiKey": "你的 API Key"  # 需在 newsapi.org 申请免费 API Key
    }
    
    try:
        # 发送 GET 请求（timeout：超时时间 10 秒）
        response = requests.get(url, params=params, timeout=10)
        
        # 检查响应状态码（200 表示成功）
        if response.status_code == 200:
            data = response.json()  # 解析 JSON 响应
            print("最新英语新闻：")
            for article in data.get("articles", [])[:3]:  # 取前 3 条
                print(f"\n标题：{article['title']}")
                print(f"来源：{article['source']['name']}")
                print(f"链接：{article['url']}")
        else:
            print(f"请求失败：状态码 {response.status_code}")
    
    except requests.exceptions.Timeout:
        print("请求超时！")
    except requests.exceptions.ConnectionError:
        print("网络连接错误！")
    except Exception as e:
        print(f"异常：{e}")

if __name__ == "__main__":
    get_english_news()
```

### 2. 简单 HTTP 服务器（内置 http.server 模块）
快速搭建静态文件服务器，用于分享英语学习资料（如单词表、PDF 文档）：
```python
import http.server
import socketserver

def start_http_server(port=8000):
    """启动简单 HTTP 服务器（分享当前目录文件）"""
    # 设置服务器地址和端口
    handler = http.server.SimpleHTTPRequestHandler
    with socketserver.TCPServer(("", port), handler) as httpd:
        print(f"HTTP 服务器已启动，访问地址：http://localhost:{port}")
        print(f"当前分享目录：{__file__[:-12]}")  # 显示当前目录
        try:
            httpd.serve_forever()  # 持续运行
        except KeyboardInterrupt:
            print("\n服务器正在关闭...")
            httpd.server_close()

if __name__ == "__main__":
    start_http_server(port=8000)
```
> 💡 运行后，在浏览器访问 `http://localhost:8000` 即可查看和下载当前目录下的英语学习资料。

## 五、高级特性：多线程与异步编程
### 1. 多线程服务器（支持高并发）
之前的 TCP 服务器已使用多线程，这里优化为线程池模式（更高效）：
```python
import socket
from concurrent.futures import ThreadPoolExecutor

word_database = {
    "perseverance": "坚持不懈，毅力",
    "consistent": "一致的，持续的",
    "proficiency": "熟练，精通"
}

def handle_client(client_socket, client_addr):
    """处理客户端请求"""
    try:
        client_socket.send("欢迎使用单词查询服务（线程池版）！输入 'quit' 退出\n".encode("utf-8"))
        while True:
            data = client_socket.recv(1024).decode("utf-8").strip()
            if not data or data.lower() == "quit":
                break
            response = f"单词：{data}\n释义：{word_database.get(data, '未找到')}\n"
            client_socket.send(response.encode("utf-8"))
    except Exception as e:
        print(f"{client_addr} 异常：{e}")
    finally:
        client_socket.close()

def start_thread_pool_server():
    """线程池 TCP 服务器（支持最大 10 个并发连接）"""
    server_socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    server_socket.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
    server_socket.bind(('', 8888))
    server_socket.listen(10)
    print("线程池 TCP 服务器已启动，监听端口 8888...")
    
    # 创建线程池（最大 10 个线程）
    with ThreadPoolExecutor(max_workers=10) as executor:
        try:
            while True:
                client_socket, client_addr = server_socket.accept()
                # 提交任务到线程池
                executor.submit(handle_client, client_socket, client_addr)
        except KeyboardInterrupt:
            print("\n服务器关闭...")
        finally:
            server_socket.close()

if __name__ == "__main__":
    start_thread_pool_server()
```

### 2. 异步网络编程（asyncio）
Python3.5+ 引入 `asyncio` 模块，支持异步 I/O，效率高于多线程（适合高并发场景）：
```python
import asyncio

word_database = {
    "perseverance": "坚持不懈，毅力",
    "consistent": "一致的，持续的",
    "proficiency": "熟练，精通"
}

async def handle_client(reader, writer):
    """异步处理客户端连接"""
    client_addr = writer.get_extra_info('peername')
    print(f"新连接：{client_addr}")
    
    try:
        # 发送欢迎消息
        welcome_msg = "欢迎使用异步单词查询服务！输入 'quit' 退出\n"
        writer.write(welcome_msg.encode("utf-8"))
        await writer.drain()  # 等待数据发送完成
        
        while True:
            # 异步读取客户端数据
            data = await reader.read(1024)
            if not data:
                break
            
            word = data.decode("utf-8").strip()
            if word.lower() == "quit":
                response = "再见！"
                writer.write(response.encode("utf-8"))
                await writer.drain()
                break
            
            # 查询单词并响应
            definition = word_database.get(word, f"未找到单词 '{word}'")
            response = f"单词：{word}\n释义：{definition}\n"
            writer.write(response.encode("utf-8"))
            await writer.drain()
    
    except Exception as e:
        print(f"{client_addr} 异常：{e}")
    finally:
        print(f"{client_addr} 断开连接")
        writer.close()
        await writer.wait_closed()

async def start_async_server():
    """启动异步 TCP 服务器"""
    server = await asyncio.start_server(
        handle_client,  # 处理函数
        host='',        # 监听所有网卡
        port=8888       # 端口
    )
    
    addr = server.sockets[0].getsockname()
    print(f"异步服务器已启动，监听 {addr}")
    
    # 持续运行服务器
    async with server:
        await server.serve_forever()

if __name__ == "__main__":
    try:
        asyncio.run(start_async_server())  # 运行异步事件循环
    except KeyboardInterrupt:
        print("\n服务器关闭...")
```

## 六、实战案例：英语学习资源下载器
结合 HTTP 请求和文件操作，实现英语文章自动下载并保存为文本文件：
```python
import requests
from bs4 import BeautifulSoup  # 需安装：pip install beautifulsoup4
import os

def download_english_articles(url, save_dir="english_articles"):
    """下载英语文章并保存为文本文件"""
    # 创建保存目录
    if not os.path.exists(save_dir):
        os.makedirs(save_dir)
    
    try:
        # 发送请求获取页面
        response = requests.get(url, timeout=10)
        response.encoding = response.apparent_encoding  # 自动识别编码
        soup = BeautifulSoup(response.text, "html.parser")
        
        # 提取文章标题和内容（根据实际网页结构调整选择器）
        title = soup.find("h1").get_text(strip=True) if soup.find("h1") else "untitled"
        content = soup.find("div", class_="article-content").get_text(strip=True, separator="\n") if soup.find("div", class_="article-content") else "无内容"
        
        # 处理标题中的特殊字符（避免文件名错误）
        invalid_chars = ['/', '\\', ':', '*', '?', '"', '<', '>', '|']
        for char in invalid_chars:
            title = title.replace(char, '_')
        
        # 保存为文本文件
        save_path = os.path.join(save_dir, f"{title}.txt")
        with open(save_path, "w", encoding="utf-8") as f:
            f.write(f"标题：{title}\n")
            f.write(f"来源：{url}\n")
            f.write("="*50 + "\n\n")
            f.write(content)
        
        print(f"✅ 下载成功！文件保存至：{save_path}")
    
    except Exception as e:
        print(f"❌ 下载失败：{e}")

if __name__ == "__main__":
    # 示例：下载简单英语文章（可替换为其他英语学习网站 URL）
    article_urls = [
        "https://www.englishsimple.com/articles/what-is-mindfulness",
        "https://www.englishsimple.com/articles/how-to-improve-english-speaking"
    ]
    
    for url in article_urls:
        download_english_articles(url)
```

## 七、常见问题与解决方案
### 1. 端口占用错误：`Address already in use`
- 解决方案 1：更换未被占用的端口（如 8889、9998）
- 解决方案 2：关闭占用端口的进程（Windows：`netstat -ano | findstr "8888"` 找到 PID 后结束进程；Linux：`lsof -i:8888` + `kill -9 PID`）
- 解决方案 3：设置端口复用（代码中 `server_socket.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)`）

### 2. 连接超时/网络不可达
- 检查服务器 IP 和端口是否正确
- 确认服务器已启动且监听对应端口
- 关闭防火墙或开放对应端口（如 Windows 防火墙、Linux iptables）
- 远程连接时确保服务器允许外部访问（如阿里云安全组开放端口）

### 3. 中文乱码问题
- 发送数据时指定编码：`data.encode("utf-8")`
- 接收数据时指定编码：`data.decode("utf-8")`
- HTTP 请求时使用 `response.encoding = response.apparent_encoding` 自动识别编码
- 文件操作时指定 `encoding="utf-8"`

### 4. 高并发场景下性能瓶颈
- 使用线程池（`concurrent.futures.ThreadPoolExecutor`）替代单线程
- 大型项目采用异步编程（`asyncio`）或框架（如 FastAPI）
- 避免在处理函数中执行耗时操作（如数据库查询、文件读写）
