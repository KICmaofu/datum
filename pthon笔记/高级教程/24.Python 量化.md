## 四、核心步骤 2：策略编写（量化的核心）
量化策略的本质是「基于数据和规则，自动生成买入/卖出信号」。常见策略包括 **技术指标策略**（MA/RSI/MACD）、**均值回归策略**、**趋势跟踪策略**、**机器学习策略** 等。以下是入门级策略示例：

### 1. 基础策略：移动平均线交叉策略（MA 金叉/死叉）
**策略逻辑**：
- 短期均线（如 5 日均线）上穿长期均线（如 20 日均线）→ 金叉 → 买入；
- 短期均线下穿长期均线 → 死叉 → 卖出；
- 仓位管理：满仓买入/清仓卖出（简化版）。

#### 用 Pandas 实现策略信号生成
```python
import pandas as pd
import matplotlib.pyplot as plt

# 1. 加载数据（以 A 股平安银行为例，已预处理）
df = pd.read_csv("000001_daily.csv")  # 假设已保存之前获取的日线数据
df["trade_date"] = pd.to_datetime(df["trade_date"])
df.set_index("trade_date", inplace=True)

# 2. 计算均线指标（5 日均线 MA5，20 日均线 MA20）
df["MA5"] = df["close"].rolling(window=5).mean()
df["MA20"] = df["close"].rolling(window=20).mean()

# 3. 生成交易信号（1=买入，-1=卖出，0=无信号）
df["signal"] = 0
# 金叉：MA5 上穿 MA20，且前一天 MA5 < MA20
df.loc[(df["MA5"] > df["MA20"]) & (df["MA5"].shift(1) < df["MA20"].shift(1)), "signal"] = 1
# 死叉：MA5 下穿 MA20，且前一天 MA5 > MA20
df.loc[(df["MA5"] < df["MA20"]) & (df["MA5"].shift(1) > df["MA20"].shift(1)), "signal"] = -1

# 4. 可视化策略信号（K 线 + 均线 + 买卖点）
plt.figure(figsize=(12, 6))
plt.plot(df["close"], label="收盘价", color="blue")
plt.plot(df["MA5"], label="MA5", color="red")
plt.plot(df["MA20"], label="MA20", color="green")

# 绘制买入信号（向上箭头）
buy_signals = df[df["signal"] == 1]
plt.scatter(buy_signals.index, buy_signals["close"], marker="^", color="red", s=100, label="买入")

# 绘制卖出信号（向下箭头）
sell_signals = df[df["signal"] == -1]
plt.scatter(sell_signals.index, sell_signals["close"], marker="v", color="green", s=100, label="卖出")

plt.legend()
plt.title("平安银行 MA 金叉死叉策略信号")
plt.show()

# 输出信号示例
print("交易信号示例：\n", df[["close", "MA5", "MA20", "signal"]].tail(10))
```

### 2. 进阶策略：RSI 超买超卖策略
**策略逻辑**：
- RSI（相对强弱指数）取值 0-100，RSI > 70 → 超买 → 卖出；
- RSI < 30 → 超卖 → 买入；
- 结合均线过滤趋势（避免震荡市误信号）。

```python
# 计算 RSI 指标（14 周期）
def calculate_rsi(series, window=14):
    delta = series.diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=window).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=window).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    return rsi

# 加载数据并计算指标
df["RSI14"] = calculate_rsi(df["close"], window=14)
df["MA10"] = df["close"].rolling(window=10).mean()

# 生成信号（超卖且价格在 MA10 上方 → 买入；超买且价格在 MA10 下方 → 卖出）
df["signal"] = 0
df.loc[(df["RSI14"] < 30) & (df["close"] > df["MA10"]), "signal"] = 1
df.loc[(df["RSI14"] > 70) & (df["close"] < df["MA10"]), "signal"] = -1

# 可视化
plt.figure(figsize=(12, 8))
# 上半部分：价格 + MA10 + 信号
plt.subplot(2, 1, 1)
plt.plot(df["close"], label="收盘价")
plt.plot(df["MA10"], label="MA10")
plt.scatter(df[df["signal"] == 1].index, df[df["signal"] == 1]["close"], marker="^", color="red", s=100)
plt.scatter(df[df["signal"] == -1].index, df[df["signal"] == -1]["close"], marker="v", color="green", s=100)
plt.legend()

# 下半部分：RSI
plt.subplot(2, 1, 2)
plt.plot(df["RSI14"], label="RSI14", color="purple")
plt.axhline(y=70, color="red", linestyle="--", label="超买线")
plt.axhline(y=30, color="green", linestyle="--", label="超卖线")
plt.legend()
plt.show()
```


## 五、核心步骤 3：策略回测（验证策略有效性）
回测是用历史数据模拟策略的交易过程，评估其收益、风险等指标（如年化收益率、最大回撤、夏普比率），避免盲目实盘。主流回测框架有 Backtrader（功能全）、VectorBT（速度快）、Vn.py（支持实盘）。

### 用 Backtrader 实现 MA 策略回测（完整示例）
```python
import backtrader as bt
import pandas as pd

# 1. 定义策略类（继承 Backtrader 的 Strategy）
class MA_Cross_Strategy(bt.Strategy):
    # 策略参数（可外部调整，方便优化）
    params = (
        ("ma_short", 5),    # 短期均线周期
        ("ma_long", 20),    # 长期均线周期
        ("print_log", True) # 是否打印日志
    )

    def __init__(self):
        # 初始化均线指标
        self.ma_short = bt.indicators.SimpleMovingAverage(
            self.data.close, period=self.params.ma_short, plotname="MA5"
        )
        self.ma_long = bt.indicators.SimpleMovingAverage(
            self.data.close, period=self.params.ma_long, plotname="MA20"
        )
        # 金叉/死叉信号（Backtrader 内置交叉指标）
        self.crossover = bt.indicators.CrossOver(self.ma_short, self.ma_long)
        # 记录当前持仓
        self.position = None

    def next(self):
        # 无持仓时，金叉买入
        if not self.position and self.crossover > 0:
            self.buy(size=1000)  # 买入 1000 股（简化仓位）
            if self.params.print_log:
                print(f"买入：日期 {self.data.datetime.date()}, 价格 {self.data.close[0]}")
        # 有持仓时，死叉卖出
        elif self.position and self.crossover < 0:
            self.sell(size=1000)  # 卖出 1000 股
            if self.params.print_log:
                print(f"卖出：日期 {self.data.datetime.date()}, 价格 {self.data.close[0]}")

    def log(self, txt, dt=None, doprint=False):
        # 日志打印函数
        if self.params.print_log and doprint:
            dt = dt or self.data.datetime.date()
            print(f"{dt}, {txt}")

# 2. 准备回测数据（适配 Backtrader 格式）
def get_backtrader_data(df):
    # 转换为 Backtrader 要求的 DataFrame 格式（datetime, open, high, low, close, volume）
    df_bt = df[["open", "high", "low", "close", "vol"]].copy()
    df_bt.index.name = "datetime"
    # 转换为 Backtrader 的数据格式
    data = bt.feeds.PandasData(
        dataname=df_bt,
        datetime=df_bt.index,
        open="open",
        high="high",
        low="low",
        close="close",
        volume="vol"
    )
    return data

# 3. 执行回测
if __name__ == "__main__":
    # 加载数据（之前获取的平安银行日线数据）
    df = pd.read_csv("000001_daily.csv")
    df["trade_date"] = pd.to_datetime(df["trade_date"])
    df.set_index("trade_date", inplace=True)
    data = get_backtrader_data(df)

    # 初始化回测引擎
    cerebro = bt.Cerebro()
    # 添加策略（可传入参数调整均线周期）
    cerebro.addstrategy(MA_Cross_Strategy, ma_short=5, ma_long=20)
    # 添加数据
    cerebro.adddata(data, name="000001.SZ")
    # 设置初始资金（10 万元）
    cerebro.broker.setcash(100000.0)
    # 设置交易成本（印花税 0.1% + 佣金 0.05%）
    cerebro.broker.setcommission(commission=0.0005, stamp_duty=0.001)
    # 设置滑点（模拟实际交易中的价格波动，0.1%）
    cerebro.broker.set_slippage_fixed(0.001)

    # 打印初始资金
    print(f"初始资金：{cerebro.broker.getvalue():.2f} 元")

    # 运行回测
    cerebro.run()

    # 打印最终资金和收益
    final_cash = cerebro.broker.getvalue()
    profit = final_cash - 100000.0
    profit_rate = (profit / 100000.0) * 100
    print(f"最终资金：{final_cash:.2f} 元")
    print(f"总收益：{profit:.2f} 元")
    print(f"收益率：{profit_rate:.2f}%")

    # 可视化回测结果（K 线、均线、买卖点、收益曲线）
    cerebro.plot(style="candlestick")
```

### 核心回测指标解读（评估策略好坏）
| 指标                | 含义                                  | 优秀标准                          |
|---------------------|---------------------------------------|-----------------------------------|
| 年化收益率          | 策略全年预期收益（复利）              | 股票策略 > 15%，期货/加密货币 > 30% |
| 最大回撤            | 策略从峰值到谷底的最大亏损比例        | 股票策略 < 20%，期货/加密货币 < 30% |
| 夏普比率            | 单位风险对应的超额收益（风险调整后收益） | > 1.5（越高越好，< 1 需谨慎）     |
| 胜率                | 盈利交易次数 / 总交易次数              | > 50%（结合盈亏比更有意义）        |
| 盈亏比              | 平均盈利 / 平均亏损                    | > 1.5（胜率低但盈亏比高也可盈利）  |


## 六、核心步骤 4：实盘交易（策略落地）

### 1. A 股实盘（Vn.py + 券商接口）
Vn.py 是国内最流行的开源量化交易框架，支持对接中信证券、华泰证券、东方财富等主流券商的接口（需开通券商量化交易权限）。

#### 核心步骤：
1. 安装 Vn.py：`pip install vnpy`；
2. 下载对应券商的交易接口插件（如 `vnpy_ctp` 对接期货、`vnpy_tiger` 对接老虎证券）；
3. 编写实盘策略（复用回测逻辑，适配 Vn.py 接口）；
4. 部署实盘（建议先小仓位测试，监控策略运行）。

#### 简化示例（Vn.py 实盘策略框架）
```python
from vnpy.app.cta_strategy import CtaTemplate, StopOrder
from vnpy.trader.object import TickData, BarData, TradeData, OrderData
from vnpy.trader.constant import Direction, Offset

class MA_Cross_CtaStrategy(CtaTemplate):
    """Vn.py 实盘版 MA 交叉策略"""
    author = "Python 量化"
    ma_short = 5
    ma_long = 20
    volume = 1000  # 交易数量

    ma_short_value = 0.0
    ma_long_value = 0.0
    long_position = 0
    short_position = 0

    parameters = ["ma_short", "ma_long", "volume"]
    variables = ["ma_short_value", "ma_long_value", "long_position", "short_position"]

    def on_bar(self, bar: BarData):
        """K 线更新时触发（实盘核心回调函数）"""
        # 计算均线
        self.ma_short_value = self.am.close.rolling(window=self.ma_short).mean().iloc[-1]
        self.ma_long_value = self.am.close.rolling(window=self.ma_long).mean().iloc[-1]

        # 更新持仓
        self.long_position = self.get_position(bar.vt_symbol).long_volume
        self.short_position = self.get_position(bar.vt_symbol).short_volume

        # 金叉买入
        if self.ma_short_value > self.ma_long_value and self.long_position == 0:
            if self.short_position > 0:
                self.close_position(bar, Direction.SHORT)  # 先平空仓
            self.buy(bar.close_price, self.volume)  # 开多仓

        # 死叉卖出
        elif self.ma_short_value < self.ma_long_value and self.short_position == 0:
            if self.long_position > 0:
                self.close_position(bar, Direction.LONG)  # 先平多仓
            self.short(bar.close_price, self.volume)  # 开空仓

    def close_position(self, bar: BarData, direction: Direction):
        """平仓函数"""
        if direction == Direction.LONG:
            self.sell(bar.close_price, self.long_position)
        elif direction == Direction.SHORT:
            self.cover(bar.close_price, self.short_position)
```

### 2. 美股/加密货币实盘（Alpaca/Binance API）
#### 加密货币实盘（Binance API，无门槛）
```python
from binance.client import Client
from binance.exceptions import BinanceAPIException, BinanceOrderException
import time

# 初始化 Binance 客户端（需在 Binance 官网创建 API 密钥）
api_key = "你的 Binance API Key"
api_secret = "你的 Binance API Secret"
client = Client(api_key, api_secret, testnet=True)  # testnet=True 表示测试网（无真实资金风险）

# 实盘策略（MA 金叉死叉，简化版）
def ma_cross_trade(symbol="BTCUSDT", interval=Client.KLINE_INTERVAL_1HOUR, ma_short=5, ma_long=20):
    while True:
        # 获取最新 K 线数据
        klines = client.get_klines(
            symbol=symbol,
            interval=interval,
            limit=ma_long + 1  # 获取足够数据计算均线
        )
        df = pd.DataFrame(klines, columns=["timestamp", "open", "high", "low", "close", "volume"])[["close"]]
        df["close"] = df["close"].astype(float)

        # 计算均线
        ma_short_val = df["close"].rolling(window=ma_short).mean().iloc[-1]
        ma_long_val = df["close"].rolling(window=ma_long).mean().iloc[-1]

        # 获取当前持仓
        balance = client.get_asset_balance(asset="USDT")
        usdt_balance = float(balance["free"])
        position = client.get_asset_balance(asset="BTC")
        btc_balance = float(position["free"])

        # 金叉买入（用 50% USDT 买入 BTC）
        if ma_short_val > ma_long_val and btc_balance < 0.001:  # 无持仓
            buy_amount = (usdt_balance * 0.5) / df["close"].iloc[-1]  # 50% 仓位
            buy_amount = round(buy_amount, 6)  #  Binance BTC 最小下单单位 0.000001
            try:
                order = client.create_order(
                    symbol=symbol,
                    side=Client.SIDE_BUY,
                    type=Client.ORDER_TYPE_MARKET,
                    quantity=buy_amount
                )
                print(f"买入成功：{order}")
            except (BinanceAPIException, BinanceOrderException) as e:
                print(f"买入失败：{e}")

        # 死叉卖出（清仓 BTC）
        elif ma_short_val < ma_long_val and btc_balance >= 0.001:  # 有持仓
            try:
                order = client.create_order(
                    symbol=symbol,
                    side=Client.SIDE_SELL,
                    type=Client.ORDER_TYPE_MARKET,
                    quantity=round(btc_balance, 6)
                )
                print(f"卖出成功：{order}")
            except (BinanceAPIException, BinanceOrderException) as e:
                print(f"卖出失败：{e}")

        # 每隔 1 小时查询一次（对应 K 线周期）
        time.sleep(3600)

# 运行实盘策略（测试网）
ma_cross_trade()
```

## 七、风险控制（量化交易的生命线）
量化交易的核心不是“赚多少钱”，而是“控制风险”。以下是必做的风险控制措施：

### 1. 仓位管理
- 固定仓位：单策略单品种持仓不超过总资金的 20%-30%；
- 凯利公式：根据策略胜率、盈亏比计算最优仓位（避免过度交易）；
- 分批建仓：避免一次性满仓，降低滑点影响。

### 2. 止损止盈
- 固定比例止损：单笔交易亏损达到 5%-10% 强制平仓；
- 移动止损：盈利后随价格上涨调整止损位（锁定利润）；
- 最大连续亏损限制：连续亏损 3-5 次后暂停交易，复盘策略。

### 3. 其他风险控制
- 滑点/交易成本：回测时必须加入真实的滑点和交易成本（否则回测收益虚高）；
- 数据质量：确保数据复权正确、无缺失值（避免策略基于错误数据）；
- 过拟合防范：避免过度优化参数（如用不同时间段的数据验证策略稳定性）；
- 监控告警：实盘时实时监控策略运行状态，异常时触发邮件/短信告警。

## 九、总结
Python 量化交易的核心流程是「数据获取→策略编写→回测验证→实盘交易→风险控制」，新手建议按以下步骤入门：
1. 掌握 Pandas/Numpy 数据处理基础；
2. 用 Tushare/AkShare 获取数据，实现简单策略（如 MA 交叉）；
3. 用 Backtrader 回测策略，优化参数和风险控制；
4. 用测试网/小仓位实盘，监控策略表现；
5. 进阶学习多因子、机器学习等复杂策略。

量化交易不是“稳赚不赔”，而是通过数据和规则降低人为情绪影响，提高交易的一致性和效率。关键是保持敬畏心，重视风险控制，持续迭代策略。
